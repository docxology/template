\begin{frame}[fragile,allowframebreaks]{Alphabetical summaries}
\protect\phantomsection\label{alphabetical-summaries}
Below, \texttt{\$\{module\_*\_python\_file\_count\}} placeholders expand
per subdirectory at render-time.

\begin{block}{\breaktt{infrastructure.autoresearch} (10 files)}
\protect\phantomsection\label{infrastructure.autoresearch-10-files}
Readiness planner, validation CLI, and report models for
AutoResearch-style project promotion
(\breaktt{infrastructure/autoresearch/}).
\end{block}

\begin{block}{\breaktt{infrastructure.benchmark} (3 files)}
\protect\phantomsection\label{infrastructure.benchmark-3-files}
Template harness scoring and comparative gate helpers exercised in CI
smoke paths.
\end{block}

\begin{block}{\breaktt{infrastructure/config} (non-package subdirectory)}
\protect\phantomsection\label{infrastructureconfig-non-package-subdirectory}
Repository-wide YAML templates and secure manifests
(\texttt{.env.template}, hardened defaults referenced by Docker + CLI).
\texttt{config/} carries no \texttt{\_\_init\_\_.py}, so it is a
configuration subdirectory rather than an importable package.
\end{block}

\begin{block}{\breaktt{infrastructure.core} (109 files)}
\protect\phantomsection\label{infrastructure.core-109-files}
Checkpointing, logging, pipeline YAML parsing, telemetry bridges,
filesystem helpers, hardened exceptions. Everything else imports logging
+ error taxonomy from here first.
\end{block}

\begin{block}{\breaktt{infrastructure.doctor} (14 files)}
\protect\phantomsection\label{infrastructure.doctor-14-files}
Checkout diagnose/fix/undo repairs for broken local workspace states.
\end{block}

\begin{block}{\breaktt{infrastructure.docker} (0 files)}
\protect\phantomsection\label{infrastructure.docker-0-files}
Pinned images / compose scaffolding for reproducible CI + remote builds.
\end{block}

\begin{block}{\breaktt{infrastructure.documentation} (12 files)}
\protect\phantomsection\label{infrastructure.documentation-12-files}
Figure registries plus glossary tooling feeding manuscript automation.
\end{block}

\begin{block}{\breaktt{infrastructure.llm} (54 files)}
\protect\phantomsection\label{infrastructure.llm-54-files}
Ollama integrations, sanitization adapters, templated reviewer flows.
\textbf{Literature ingestion now lives primarily in
\breaktt{search/literature} + citation helpers in \texttt{reference/}.}
\end{block}

\begin{block}{\breaktt{infrastructure.methods} (5 files)}
\protect\phantomsection\label{infrastructure.methods-5-files}
Deterministic methods-orchestration contracts (\texttt{MethodStage},
\breaktt{MethodsOrchestrationPlan}, \texttt{MethodsIssue}): builds and
validates an ordered methods plan for a research project so the
manuscript's ``Methods'' track stays bound to executable stages.
\end{block}

\begin{block}{\breaktt{infrastructure.orchestration} (8 files)}
\protect\phantomsection\label{infrastructure.orchestration-8-files}
\breaktt{python\ -m\ infrastructure.orchestration} exposes interactive
menus, subprocess wiring for thin shell wrappers (\texttt{run.sh},
\texttt{secure\_run.sh}), and stubs used in CI for menu parsing tests.
\end{block}

\begin{block}{\breaktt{infrastructure.project} (27 files)}
\protect\phantomsection\label{infrastructure.project-27-files}
Canonical discovery (\breaktt{discover\_projects}) enforcing
\texttt{src/} + \texttt{tests/}, slug validation, nested WIP namespaces.
\end{block}

\begin{block}{\breaktt{infrastructure.prose} (9 files)}
\protect\phantomsection\label{infrastructure.prose-9-files}
Readability metrics + Markdown tooling for prose-centric manuscripts /
CI gates.
\end{block}

\begin{block}{\breaktt{infrastructure.publishing} (71 files)}
\protect\phantomsection\label{infrastructure.publishing-71-files}
Metadata models, APA/BibTeX/MLA formatters, optional Zenodo clients.
\end{block}

\begin{block}{\breaktt{infrastructure.reference} (16 files)}
\protect\phantomsection\label{infrastructure.reference-16-files}
Citation/BibTeX parsing + conversion utilities leveraged by manuscripts
and retrieval scripts.
\end{block}

\begin{block}{\breaktt{infrastructure.rendering} (50 files)}
\protect\phantomsection\label{infrastructure.rendering-50-files}
Pandoc shim, Unicode/XeLaTeX postprocessors, combined PDF/HTML/slide
exporters.
\end{block}

\begin{block}{\breaktt{infrastructure.reporting} (57 files)}
\protect\phantomsection\label{infrastructure.reporting-57-files}
Parses pytest + coverage artefacts for dashboards; pairs with Stage~01
summaries and downstream executive exports.
\end{block}

\begin{block}{\breaktt{infrastructure.scientific} (4 files)}
\protect\phantomsection\label{infrastructure.scientific-4-files}
Stability probing, benchmarking hooks---consumed heavily by optimization
exemplars (\breaktt{template\_code\_project} scripts).
\end{block}

\begin{block}{\breaktt{infrastructure.search} (44 files)}
\protect\phantomsection\label{infrastructure.search-44-files}
\texttt{literature/} client stack (\texttt{client.py}, backends, caches)
powering archive-only \breaktt{template\_search\_project} literature
workflows when copied locally from \breaktt{projects/archive/}.
\end{block}

\begin{block}{\breaktt{infrastructure.sia} (10 files)}
\protect\phantomsection\label{infrastructure.sia-10-files}
Generic Self-Improving-AI loop utilities: task-layout validation,
execution harness, and metric capture reused by \texttt{template\_sia}
(fixture-replay by default).
\end{block}

\begin{block}{\breaktt{infrastructure.skills} (7 files)}
\protect\phantomsection\label{infrastructure.skills-7-files}
Discovers \texttt{SKILL.md} frontmatter →
\breaktt{.cursor/skill\_manifest.json}.
\end{block}

\begin{block}{\breaktt{infrastructure.steganography} (11 files)}
\protect\phantomsection\label{infrastructure.steganography-11-files}
Watermark overlays, hashing companions triggered by secure pipeline
path.
\end{block}

\begin{block}{\breaktt{infrastructure.validation} (83 files)}
\protect\phantomsection\label{infrastructure.validation-83-files}
Markdown + PDF + integrity CLIs underpinning Stage~04 diagnostics.
\end{block}

\begin{block}{\breaktt{infrastructure/logrotate.d} (0 files)}
\protect\phantomsection\label{infrastructurelogrotate.d-0-files}
Operational templates for deployments (documentation-first;
intentionally minimal Python footprint).
\end{block}
\end{frame}

by FAIRsoft guidance \breakseq{[@garijo2024fairsoft]}.
