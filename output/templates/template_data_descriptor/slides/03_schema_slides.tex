\section{Schema and data dictionary}\label{sec:schema}

\begin{frame}[fragile,allowframebreaks]{Schema and data dictionary}
The descriptor's \texttt{fields} list is the dataset's \textbf{data
dictionary}: it names each column of the primary measurement table and
declares its type, nullability, optional unit, and value constraints.
\breakseq{[@fig:schema\_overview]} renders this dictionary directly from the
descriptor via \breaktt{schema\_table\_rows()} in
\breaktt{src/data\_descriptor/figures.py}.

\begin{figure}
\centering
\pandocbounded{\includegraphics[keepaspectratio,alt={Field schema / data dictionary: one row per declared field, with type, nullability, unit, and a compact constraint label, from schema\_table\_rows(). Identifier fields carry regular-expression patterns; the categorical fields carry closed enumerations; the quantitative field carries a unit and numeric bounds. The table is generated from data/example\_descriptor.json and cannot silently disagree with it.}]{../figures/schema_overview.png}}
\caption{Field schema / data dictionary: one row per declared field,

The table is generated from

\end{figure}
\end{frame}

\begin{frame}[fragile,allowframebreaks]{Field contract}
\protect\phantomsection\label{field-contract}
The six declared fields exercise the constraint vocabulary the validator
understands:

\begin{itemize}
\tightlist
\item
  \textbf{\texttt{sample\_id}} (\texttt{string}, not null) --- the
  primary key, constrained to the pattern
  \texttt{\^{}S{[}0-9{]}\{3\}\$}.
\item
  \textbf{\texttt{subject\_id}} (\texttt{string}, not null) --- foreign
  key into \breaktt{fixtures/subjects.csv}, constrained to
  \texttt{\^{}P{[}0-9{]}\{3\}\$}.
\item
  \textbf{\texttt{group}} (\texttt{category}, not null) --- a closed
  enumeration of \texttt{control} and \texttt{treatment}.
\item
  \textbf{\texttt{value}} (\texttt{number}, not null) --- the
  measurement, carrying the unit \breaktt{normalized\_score} and numeric
  bounds \texttt{{[}0,\ 1{]}}.
\item
  \textbf{\texttt{collected\_on}} (\texttt{date}, not null) --- an
  ISO-8601 collection date.
\item
  \textbf{\texttt{instrument}} (\texttt{category}, not null) --- a
  closed enumeration of \texttt{sensor\_a} and \texttt{sensor\_b}.
\end{itemize}
\end{frame}

\begin{frame}[fragile,allowframebreaks]{Schema fingerprint}
\protect\phantomsection\label{schema-fingerprint}
\breaktt{descriptor\_fingerprint()} reduces the field list to a stable,
order-independent sha256 fingerprint over
\breaktt{(name,\ type,\ nullable)} triples. Reordering the fields leaves
the fingerprint unchanged, so two descriptors with the same schema hash
identically regardless of authoring order --- the property is asserted
directly in the test suite. The fingerprint is what lets a release
manifest reference ``the schema'' by a single value that changes if and
only if the field contract changes.
\end{frame}

\begin{frame}[allowframebreaks]{Constraint validation}
\protect\phantomsection\label{constraint-validation}
The validator flags quantitative fields that lack a unit, categorical
fields that lack allowed values, enumerations that are present but
empty, numeric bounds where the minimum exceeds the maximum, and
patterns declared on non-text fields. On the shipped descriptor none of
these fire --- the data dictionary above is complete, so the readiness
score is unpenalised (see \breakseq{[@sec:quality]}).
\end{frame}
